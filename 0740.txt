    // === 10. ì‹ ê·œ: ê²°ê³¼ ê³„ì‚° ë¡œì§ (ìˆ˜ì • ì—†ìŒ) ===
    function calculateAndDisplayResults(event) {
      event.preventDefault();

      const form = event.target;
      const jobGroupSelect = document.getElementById('q_job_group');
      const jobGroupValue = jobGroupSelect.value;
      const jobGroupText = jobGroupSelect.options[jobGroupSelect.selectedIndex].text;

      // ì§ë¬´ ê·¸ë£¹ ìœ íš¨ì„± ê²€ì‚¬
      if (!jobGroupValue) {
        alert('Q0. ì§ë¬´ ê·¸ë£¹ì„ ì„ íƒí•´ ì£¼ì„¸ìš”.');
        jobGroupSelect.scrollIntoView({ behavior: 'smooth', block: 'center' });
        jobGroupSelect.style.border = '2px solid red';
        setTimeout(() => { jobGroupSelect.style.border = '1px solid #a5b4fc'; }, 2000);
        return;
      }

      let allAnswered = true;
      let scores = { I: [], II: [], III: [], IV: [] };
      userResponses = [];

      // 25ê°œ ë¬¸í•­ ìœ íš¨ì„± ê²€ì‚¬
      for (let i = 1; i <= 25; i++) {
        const qName = `q${i}`;
        const checkedInput = form.querySelector(`input[name="${qName}"]:checked`);
        
        if (!checkedInput) {
          allAnswered = false;
          const questionElement = document.querySelector(`input[name="${qName}"]`).closest('div[class*="pb-4"]');
          if(questionElement) {
            questionElement.scrollIntoView({ behavior: 'smooth', block: 'center' });
            questionElement.style.border = '2px solid red';
            questionElement.style.borderRadius = '0.5rem';
            setTimeout(() => {
              questionElement.style.border = '';
              questionElement.style.borderRadius = '';
            }, 2000);
          }
          break;
        }

        const score = parseInt(checkedInput.value);
        const areaId = checkedInput.getAttribute('data-area');
        const questionContent = checkedInput.getAttribute('data-q-content');
        
        userResponses.push({
          index: i,
          area: areaId,
          content: questionContent,
          score: score
        });

        scores[areaId].push(score);
      }

      if (!allAnswered) {
        alert('ë¯¸ì‘ë‹µ ë¬¸í•­ì´ ìˆìŠµë‹ˆë‹¤. ì‘ë‹µë˜ì§€ ì•Šì€ ë¬¸í•­ìœ¼ë¡œ ì´ë™í•©ë‹ˆë‹¤.'); 
        return;
      }
      
      let results = {
        jobGroup: { key: jobGroupValue, text: jobGroupText }
      };
      
      const sumI = scores.I.reduce((acc, val) => acc + val, 0);
      const sumII = scores.II.reduce((acc, val) => acc + val, 0);
      const sumIII = scores.III.reduce((acc, val) => acc + val, 0);
      const sumIV = scores.IV.reduce((acc, val) => acc + val, 0);
      
      const totalScore = sumI + sumII + sumIII + sumIV; // 125ì  ë§Œì 
      const averageScore = parseFloat((totalScore / 25).toFixed(2)); // 5ì  ë§Œì 
      
      const WHARTON_AVG = 3.06;
      const WHARTON_STD_DEV = 0.6;
      const zScore = parseFloat(((averageScore - WHARTON_AVG) / WHARTON_STD_DEV).toFixed(2));
      const percentileRank = getPercentileRange(zScore);
      const overallLevel = getOverallLevel(totalScore);

      results.I = {
        sum: sumI,
        max: 30,
        feedback: feedbackData.I[getFeedbackLevel_30(sumI)]
      };
      results.II = {
        sum: sumII,
        max: 35,
        feedback: feedbackData.II[getFeedbackLevel_35(sumII)]
      };
      results.III = {
        sum: sumIII,
        max: 30,
        feedback: feedbackData.III[getFeedbackLevel_30(sumIII)]
      };
      results.IV = {
        sum: sumIV,
        max: 30,
        feedback: feedbackData.IV[getFeedbackLevel_30(sumIV)]
      };

      results.totalScore = totalScore;
      results.averageScore = averageScore;
      results.zScore = zScore;
      results.percentileRank = percentileRank;
      results.overallLevel = overallLevel;

      currentResults = results;
      sendResultsToSheets(results, userResponses); 
    }
